# Email 2FA Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add mandatory email OTP (6-digit, 5-minute TTL) as a second factor for all web logins, with brute-force protection, resend cooldown, and full audit trail.

**Architecture:** Password step is verified manually in `loginAction` (outside Auth.js); on success a signed `otpPending` HttpOnly cookie is set and the user is redirected to `/login/otp`. After OTP verification, a short-lived signed `preAuthToken` is passed to `signIn("credentials", ...)` which creates the real session. Auth.js credentials provider is reduced to only accepting preAuthTokens — password path removed to prevent 2FA bypass.

**Tech Stack:** Next.js 16 (App Router) · Auth.js v5 · Prisma · `jose` (JWT/HMAC, transitive via next-auth, added as direct dep) · `nodemailer` (SMTP) · Argon2id (via `@node-rs/argon2`, existing) · Vitest · Playwright

```ts
import "server-only";
import { prisma } from "@/lib/prisma";
import { hashPassword, verifyPassword } from "./password";

const OTP_TTL_MS = 5 * 60_000;
const MAX_ATTEMPTS = 3;
const COOLDOWN_MS = 60_000;

export function generateOtpCode(): string {
  const testCode = process.env.OTP_TEST_CODE;
  if (testCode && process.env.NODE_ENV !== "production") return testCode;
  return String(Math.floor(Math.random() * 1_000_000)).padStart(6, "0");
}

export function hashOtpCode(code: string): Promise<string> {
  return hashPassword(code);
}

export async function createEmailOtp(userId: string): Promise<string> {
  const code = generateOtpCode();
  const codeHash = await hashOtpCode(code);
  const expiresAt = new Date(Date.now() + OTP_TTL_MS);
  await prisma.$transaction([
    prisma.emailOtp.updateMany({ where: { userId, usedAt: null }, data: { usedAt: new Date() } }),
    prisma.emailOtp.create({ data: { userId, codeHash, expiresAt } }),
    prisma.auditLog.create({ data: { userId, action: "otp.sent", level: "info" } }),
  ]);
  return code;
}

export async function verifyEmailOtp(
  userId: string,
  code: string,
): Promise<"ok" | "invalid" | "expired" | "max_attempts"> {
  const otp = await prisma.emailOtp.findFirst({
    where: { userId, usedAt: null },
    orderBy: { createdAt: "desc" },
  });

  if (!otp) return "invalid";

  if (otp.expiresAt < new Date()) {
    await prisma.auditLog.create({ data: { userId, action: "otp.expired", level: "warn" } });
    return "expired";
  }

  if (otp.attempts >= MAX_ATTEMPTS) return "max_attempts";

  const valid = await verifyPassword(otp.codeHash, code);
  if (!valid) {
    const newAttempts = otp.attempts + 1;
    await prisma.$transaction([
      prisma.emailOtp.update({ where: { id: otp.id }, data: { attempts: newAttempts } }),
      prisma.auditLog.create({ data: { userId, action: "otp.failed", level: "warn" } }),
    ]);
    if (newAttempts >= MAX_ATTEMPTS) return "max_attempts";
    return "invalid";
  }

  await prisma.$transaction([
    prisma.emailOtp.update({ where: { id: otp.id }, data: { usedAt: new Date() } }),
    prisma.auditLog.create({ data: { userId, action: "otp.verified", level: "info" } }),
  ]);
  return "ok";
}

export async function resendOtp(
  userId: string,
): Promise<{ type: "ok"; code: string } | { type: "cooldown"; remainingSeconds: number }> {
  const latest = await prisma.emailOtp.findFirst({
    where: { userId },
    orderBy: { createdAt: "desc" },
  });

  if (latest) {
    const elapsed = Date.now() - latest.createdAt.getTime();
    if (elapsed < COOLDOWN_MS) {
      return { type: "cooldown", remainingSeconds: Math.ceil((COOLDOWN_MS - elapsed) / 1000) };
    }
  }

  const code = await createEmailOtp(userId);
  return { type: "ok", code };
}
```

```ts
import "server-only";
import { SignJWT, jwtVerify } from "jose";
import { cookies } from "next/headers";
import type { RoleCode } from "@/lib/types/roles";
import { findUserById } from "./users";

const COOKIE_NAME = "otp-pending";
const COOKIE_MAX_AGE = 10 * 60; // 10 min — enough for OTP flow

function getSecret(): Uint8Array {
  const s = process.env.AUTH_SECRET;
  if (!s) throw new Error("AUTH_SECRET not set");
  return new TextEncoder().encode(s);
}

export async function setOtpPendingCookie(userId: string): Promise<void> {
  const token = await new SignJWT({ userId })
    .setProtectedHeader({ alg: "HS256" })
    .setIssuedAt()
    .setExpirationTime("10m")
    .sign(getSecret());

  const store = await cookies();
  store.set(COOKIE_NAME, token, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "strict",
    maxAge: COOKIE_MAX_AGE,
    path: "/",
  });
}

export async function readOtpPendingCookie(): Promise<{ userId: string } | null> {
  const store = await cookies();
  const raw = store.get(COOKIE_NAME)?.value;
  if (!raw) return null;
  try {
    const { payload } = await jwtVerify(raw, getSecret());
    const userId = payload.userId;
    if (typeof userId !== "string") return null;
    return { userId };
  } catch {
    return null;
  }
}

export async function clearOtpPendingCookie(): Promise<void> {
  const store = await cookies();
  store.delete(COOKIE_NAME);
}

// --- Pre-auth token (one-time, 2-min TTL, consumed by credentials.ts authorize) ---

export async function generatePreAuthToken(userId: string): Promise<string> {
  return new SignJWT({ type: "otp-pre-auth" })
    .setProtectedHeader({ alg: "HS256" })
    .setSubject(userId)
    .setIssuedAt()
    .setExpirationTime("2m")
    .sign(getSecret());
}

export async function verifyPreAuthToken(
  token: string,
): Promise<{ id: string; name: string; email: string; role: RoleCode } | null> {
  try {
    const { payload } = await jwtVerify(token, getSecret());
    if (payload.type !== "otp-pre-auth") return null;
    const userId = payload.sub;
    if (!userId) return null;
    const user = await findUserById(userId);
    if (!user || user.disabled) return null;
    return { id: user.id, name: user.name, email: user.email, role: user.role };
  } catch {
    return null;
  }
}

"use server";

import { z } from "zod";
import { redirect } from "next/navigation";
import { findUserByEmail } from "@/lib/auth/users";
import { isLocked, recordFailure } from "@/lib/auth/lockout";
import { verifyPassword } from "@/lib/auth/password";
import { createEmailOtp } from "@/lib/auth/otp";
import { setOtpPendingCookie } from "@/lib/auth/otp-cookie";
import { sendOtpEmail } from "@/lib/mailer";

const inputSchema = z.object({
  email: z.string().email(),
  password: z.string().min(1),
});

export interface LoginState {
  error?: boolean;
}

export async function loginAction(
  _prev: LoginState | undefined,
  formData: FormData,
): Promise<LoginState> {
  const parsed = inputSchema.safeParse({
    email: String(formData.get("email") ?? ""),
    password: String(formData.get("password") ?? ""),
  });
  if (!parsed.success) return { error: true };

  const { email, password } = parsed.data;

  const user = await findUserByEmail(email);
  if (!user || user.disabled || isLocked(user.lockedUntil)) {
    await recordFailure(user?.id ?? null, email);
    return { error: true };
  }

  const ok = await verifyPassword(user.passwordHash, password);
  if (!ok) {
    await recordFailure(user.id, email);
    return { error: true };
  }

  // Password correct — issue OTP (recordSuccess happens after OTP verification)
  const code = await createEmailOtp(user.id);
  await sendOtpEmail(user.email, code);
  await setOtpPendingCookie(user.id);

  redirect("/login/otp");
}
```

```ts
"use server";

import { AuthError } from "next-auth";
import { signIn } from "@/auth";
import { findUserById } from "@/lib/auth/users";
import { verifyEmailOtp, resendOtp } from "@/lib/auth/otp";
import { readOtpPendingCookie, clearOtpPendingCookie, generatePreAuthToken } from "@/lib/auth/otp-cookie";
import { recordSuccess } from "@/lib/auth/lockout";
import { sendOtpEmail } from "@/lib/mailer";

export type OtpError =
  | "errorInvalid"
  | "errorExpired"
  | "errorMaxAttempts"
  | "errorSessionExpired";

export interface OtpState {
  error?: OtpError;
}

export interface ResendResult {
  ok?: boolean;
  cooldown?: number;
  error?: string;
}

export async function verifyOtpAction(
  _prev: OtpState | undefined,
  formData: FormData,
): Promise<OtpState> {
  const code = String(formData.get("code") ?? "").trim();

  const pending = await readOtpPendingCookie();
  if (!pending) return { error: "errorSessionExpired" };

  const result = await verifyEmailOtp(pending.userId, code);
  if (result === "expired") return { error: "errorExpired" };
  if (result === "max_attempts") return { error: "errorMaxAttempts" };
  if (result === "invalid") return { error: "errorInvalid" };

  const user = await findUserById(pending.userId);
  if (!user) return { error: "errorSessionExpired" };

  await recordSuccess(user.id, user.email);
  await clearOtpPendingCookie();

  const preAuthToken = await generatePreAuthToken(user.id);

  try {
    await signIn("credentials", { preAuthToken, redirectTo: "/dashboard" });
    return {};
  } catch (error) {
    if (error instanceof AuthError) return { error: "errorSessionExpired" };
    throw error; // redirect propagates
  }
}

export async function resendOtpAction(_prev: ResendResult | undefined, _formData: FormData): Promise<ResendResult> {
  const pending = await readOtpPendingCookie();
  if (!pending) return { error: "session_expired" };

  const result = await resendOtp(pending.userId);
  if (result.type === "cooldown") return { cooldown: result.remainingSeconds };

  const user = await findUserById(pending.userId);
  if (!user) return { error: "session_expired" };

  await sendOtpEmail(user.email, result.code);
  return { ok: true };
}
```
